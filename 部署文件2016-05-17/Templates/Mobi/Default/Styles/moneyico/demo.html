    url('moneyico.woff') format('woff'), /* chrome、firefox */
    url('moneyico.ttf') format('truetype'), /* chrome、firefox、opera、Safari, Android, iOS 4.2+*/
    url('moneyico.svg#moneyico') format('svg'); /* iOS 4.1- */
}
